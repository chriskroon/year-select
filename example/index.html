Step 2. Add HTML and JavaScript to use

Default options 
                
              
                $('input.default[type="year"]').click(function(e){
                  e.preventDefault()
                  showYear($(this),{})
                })
                $('input.default[type="year"]').keydown(function(e){
                  e.preventDefault()
                })
Custom options 
                
              
                $('input.custom-options[type="year"]').click(function(e){
                  e.preventDefault()
                  showYear($(this),
                    {
                      years_before_now: 5,
                      years_after_now: 100
                    }
                  )
                })
                $('input.custom-options[type="year"]').keydown(function(e){
                  e.preventDefault()
                })
